        uint32_t pack(uchar bits[32]) {
            uint32_t packed = 0;
            for (uchar i = 0; i < 32; i++) {
                packed |= (uint32_t) (bits[i] >> 7) << (31 - i);
            }
            return packed;
        }
    \end{minted}

        void unpack(uchar result[32], uint32_t packed) {
            for (uchar i = 0; i < 32; i++) {
                result[i] = ((packed >> (31 - i)) & 0x1) * 255;
                }
            }
        }
